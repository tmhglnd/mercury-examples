// OldSk0Ol Dubst3p
// Creating some low tempo late 2000's dubstep with basic back-beat and a 
// wobble bass synth. The synth is made of a square wave with a modulated
// filter. Some notes play every 2 bars, and the distortion effect alternates
// between some more wet and some more dry sound.

set scale minor g
set tempo 72

// The kick plays every 1/2, but one gets delayed by an 1/8 note with wait
new sample kick_vintage time(1/2) wait([0 1/8])
// The snare is created from a metal sound with some degrade and reverb effects
new sample metal_lo name(snare) time(1/2 1/4) speed(1.5) shape(1 1/8 1) 
    set snare fx(degrade 0.5 0.8) fx(reverb 0.2 1)
// The hihat is fairly basic, just a bit shorter with the shape
new sample hat_dub time(1/16) speed(1.4) shape(1 1/32) gain(0.4)

// The wobble bass is created from a square wave oscillator
new synth square name(bass)
    // The notes play every 2 bars, and are rotated to the end of the bar, the bass plays in 1/16 notes
    set bass note(rotate(every([5 3 3 2 2] 2) -5) -1) time(1/16) shape(1 1/1 1) 
    // The filter modulates between 80 and 300/1200 Hz, the high part slowly moves from 300-1200 over 128 steps with a sinewave
    // The distortion (fuzz) alternates between 0.5 and 0.1 wet, removing the distortion sometimes
    set bass fx(filter low 1/16 80 sine(128 1 300 1200) ) fx(fuzz 15 repeat([0.5 0.1] [8 8]))

set all meter()
